create table players (player_id int not null auto_increment, name varchar(50), age int, player_type varchar(15), ig_followers int, fb_followers int, base_price int, strike_rate float, no_of_wickets int, economy float, is_sold bool, bidding_round int, team_id int, primary key (player_id));


create table players_login(id int not null auto_increment, username varchar(20) not null, password varchar(20) not null, player_id int, primary key(id), foreign key (player_id) references players(player_id), unique(username));


create table teams(team_id int not null auto_increment, team_owner varchar(50), team_name varchar(50), primary key(team_id));


create table team_login(id int not null auto_increment, login_id varchar(20) not null, password varchar(20), team_id int, primary key(id), foreign key(team_id) references teams(team_id), unique(login_id));
